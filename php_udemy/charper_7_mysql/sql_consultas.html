Temas básicos de SQl

CREATE DATABASE + nombre de la base a crear; Para crear bases de datos 
		
			SHOW DATABASES;
Para ver la bases de datos que existen
Crear la tabla con el nombre y los campos además de las caracteristicas de dichos campos. 
		
		CREATE TABLE test (
			id INT AUTO_INCREMENT, # nombre del campo, tipo y auto incremento
			nombre VARCHAR (100) NOT NULL, # nombre del campo, tipo, longitud y obligatorieda
			mail VARCHAR NOT NULL, # nombre del campo, tipo, longitud y obligatorieda
			PRIMARY KEY (id)  #campo que va a ser la llave primaria
		);
DESCRIBE + nombre de la tabla # Ver las columnas y descrición de las tablas 
		
		DESCRIBE users;
		
		
		
		SELECT * FROM users;
INSERT INTO + nombre de la tabla VALUES(los valores a ingresar en el orden de la tabla);
Cuando hay un campo marcado como AUTO_INCREMENT se debe ingresar como NULL para que él solo tomé el valor 
		
		INSERT INTO users
		VALUES
			(
				NULL,
				'anastacio',
				'anastacio@gmail.com'
			);
En caso de solo querer ingresar datos a ciertos campos de la tabla utilizamos el siguiente comando
INSERT INTO nombre de la tabla (campos a llenar separados por comas) VALUES (valores de los campos separados por comas); 
		
		INSERT INTO users (id, mail)
		VALUES
			(NULL, 'test@test.com'),
		    (NULL, 'test_dos@test.com')
		;
Para actuzalizar los valores de la tablas se usa en comando UPDATE
UPDATE nombre de la tabla SET nombre del campo a modificar = nuevo valor WHERE campo identificador = valor del campo identificador 
		
		UPDATE users SET name='Anastacio' where id = 7; 
		
		
		
		UPDATE users SET name='Carlos' where id = 5;
		
		
		
		UPDATE users SET mail='maniatik@gmail.com' where id = 2;
consultas de un solo campo o campos especificos
SELECT nombre o nombres de los campos FROM tabla a consultar; 
		
		SELECT name FROM users;
El WHERE es una buena opción para buscar en base a un campo 
		
		SELECT name FROM users where `name`= 'Juank';
LIKE es simular al WHERE pero tiene más funcinalidades
En este ejemplo busca todo lo que empieza por A 
		
		SELECT * FROM users WHERE `name` LIKE 'c%';
En este ejemplo busca todo lo que termina en k el signo % indica todo lo que este adelante o atrás de él 
		
		SELECT * FROM users WHERE `name` LIKE '%k';
		
		
		
		SELECT * FROM users WHERE mail LIKE '%.co%';
ORDER BY sirve para ordenar segun un campo y ASC y DESC pueden estar unidos a el 
		
		SELECT * FROM users ORDER BY `name` ASC;
		
		
		
		SELECT * FROM users ORDER BY `name` DESC;
El comado LIMIT sirve para indicar que no traiga todos los resultados si no los que se indiquen despues del comando
LIMIT con estos parámetros indica un rago LIMIT 0, 2 indica que traiga dos valores despues del primer registro 
		
		SELECT * FROM users LIMIT 0, 3;
		
		
		
		SELECT * FROM users;
DELETE es la palabra reservada para eliminar registros se indica la tabla y el WHERE para especificar que se va a borrar; 
		
		DELETE FROM users WHERE id = 6;
